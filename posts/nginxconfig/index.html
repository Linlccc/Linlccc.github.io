# 老域名跳转新域名
server {
  listen 80;
  server_name old.cxk.cn;
  location / {
    rewrite ^/(.*)$ https://new.cxk.cn/$1;
  }
}

# 路径重定向
server {
  listen 80;
  server_name old.cxk.cn;
  location / {
    rewrite ^/old.cxk.cn/(.*)$ /new-path/$1;
  }
}
